
// Problem: 3075. Maximize Happiness of Selected Children
// URL: https://leetcode.com/problems/maximize-happiness-of-selected-children/?envType=daily-question&envId=2026-01-02

class Solution {
    public long maximumHappinessSum(int[] happiness, int k) {
        Arrays.sort(happiness);
        long result = 0;
        int count = 0;
        for(int i = happiness.length - 1; i >= 0 ; i--){
            if(count == k){
                break;
            }
            result += ((happiness[i] - count)) <= 0 ? 0 : (happiness[i] - count) ;
            count++;
        }
        return result;
    }
}
