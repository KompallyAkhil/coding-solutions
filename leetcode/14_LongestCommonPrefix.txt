// Problem: 14. Longest Common Prefix
//

// Solution:
class Solution {
    public String longestCommonPrefix(String[] strs) {
        String maxi = Collections.max(Arrays.asList(strs));
        String mini = Collections.min(Arrays.asList(strs));
        String result = "";
        int i = 0;
        int j = 0;
        while( i < maxi.length() && j < mini.length()){
            if (maxi.charAt(i) == mini.charAt(j)){
                result += maxi.charAt(i);
                i++;
                j++;
            }
            else{
                break;
            }
        }
        return result;
    }
}
